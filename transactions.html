No ledger activity matches the selected filters.
`; return; } $("activityList").innerHTML = groups.map((group, index) => { const isOpen = state.openGroupId === group.id; const subtitle = [ group.date, group.accounts.join(" → "), group.categories.filter((x) => x && x !== "Uncategorized").join(", ") ].filter(Boolean).join(" · "); const status = group.hasReversal ? pill("Reversal linked", "badge-warning") : group.isGrouped ? pill("Grouped", "badge-ready") : pill("Single", "badge-unknown"); const intl = group.kind === "international" ? pill("International package", "badge-info") : ""; const merchant = group.hasMerchant ? pill("Merchant", "badge-info") : ""; const reverseButton = group.reverse.allowed ? `` : ``; return `
${pill(group.kind, "badge-info")} ${status} ${intl} ${merchant}
${escapeHtml(group.title)}
${escapeHtml(subtitle)}
Click to open details and actions
${escapeHtml(amountSign(group))} ${money(group.displayAmount)}

`; }).join(""); }
